# GitHub 仓库同步工具 - 项目架构文档

## 一、项目概述

### 1.1 项目目标
构建一个基于 Python + PyQt 的桌面应用程序，实现本地文件夹与 GitHub 仓库的关联管理，提供代码同步、变更追踪、日志管理等功能。

### 1.2 核心价值
- 简化 Git 操作流程，降低使用门槛
- 自动化日志生成与同步
- 多项目统一管理
- 可视化状态监控

# 使用事件总线解耦模块
class EventBus:
    def __init__(self):
        self._handlers: Dict[str, List[Callable]] = {}
    
    def subscribe(self, event_type: str, handler: Callable):
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)
    
    def publish(self, event_type: str, data: Any):
        for handler in self._handlers.get(event_type, []):

#### 3. 实现命令模式支持撤销
```python
class Command(ABC):
    @abstractmethod
    def execute(self) -> Result: ...
    
    @abstractmethod
    def undo(self) -> Result: ...

class CommitCommand(Command):
    def __init__(self, git_engine: GitEngine, message: str):
        self.git_engine = git_engine
        self.message = message
        self.commit_hash: Optional[str] = None
    
    def execute(self) -> Result:
        result = self.git_engine.commit(self.message)
        self.commit_hash = result.data.get("hash")
        return result
    
    def undo(self) -> Result:
        if self.commit_hash:
            return self.git_engine.reset(self.commit_hash, mode="soft")
        return Result.error("No commit to undo")

class CommandManager:
    def __init__(self, max_history: int = 50):
        self._history: List[Command] = []
        self._max_history = max_history
    
    def execute(self, command: Command) -> Result:
        result = command.execute()
        if result.success:
            self._history.append(command)
            if len(self._history) > self._max_history:
                self._history.pop(0)
        return result
    
    def undo(self) -> Result:
        if not self._history:
            return Result.error("No command to undo")
        return self._history.pop().undo()
```

class IncrementalSync:
    def __init__(self, git_engine: GitEngine):
        self.git_engine = git_engine
    
    async def sync(self, repo_id: str) -> SyncResult:
        last_commit = await self._get_last_synced_commit(repo_id)
        current_commit = await self.git_engine.get_head_commit()
        
        if last_commit == current_commit:
            return SyncResult(status="no_changes")
        
        changes = await self.git_engine.get_commit_range(
            last_commit, current_commit
        )
        
        return SyncResult(
            status="success",
            changes=changes,
            commits_count=len(changes)
        )
```

class UpdateManager:
    def __init__(self, current_version: str, update_url: str):
        self.current_version = current_version
        self.update_url = update_url
    
    async def check_update(self) -> Optional[UpdateInfo]:
        async with aiohttp.ClientSession() as session:
            async with session.get(f"{self.update_url}/latest") as resp:
                data = await resp.json()
        
        latest = data["version"]
        if self._compare_versions(latest, self.current_version) > 0:
            return UpdateInfo(
                version=latest,
                download_url=data["download_url"],
                changelog=data["changelog"],
                mandatory=data.get("mandatory", False)
            )
        return None
    
    async def download_update(self, update_info: UpdateInfo, progress_callback: Callable):
        async with aiohttp.ClientSession() as session:
            async with session.get(update_info.download_url) as resp:
                total = int(resp.headers.get('content-length', 0))
                downloaded = 0
                
                with tempfile.NamedTemporaryFile(delete=False, suffix='.exe') as f:
                    async for chunk in resp.content.iter_chunked(8192):
                        f.write(chunk)
                        downloaded += len(chunk)
                        progress = (downloaded / total) * 100
                        progress_callback(progress)
                    
                    return f.name
```
